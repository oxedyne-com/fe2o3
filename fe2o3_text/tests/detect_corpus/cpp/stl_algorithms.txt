#include <algorithm>
#include <numeric>
#include <vector>
#include <string>
#include <map>
#include <iostream>

struct Student {
    std::string name;
    int grade;
    double score;
};

int main() {
    std::vector<Student> students = {
        {"Alice",   12, 92.5},
        {"Bob",     11, 87.3},
        {"Charlie", 12, 91.0},
        {"Diana",   11, 88.7},
        {"Eve",     12, 95.2},
    };

    // Sort by score descending.
    std::sort(students.begin(), students.end(),
        [](const Student& a, const Student& b) {
            return a.score > b.score;
        });

    std::cout << "ranking:\n";
    for (const auto& s : students) {
        std::cout << "  " << s.name << ": " << s.score << "\n";
    }

    // Average score.
    double total = std::accumulate(students.begin(), students.end(), 0.0,
        [](double sum, const Student& s) { return sum + s.score; });
    std::cout << "average: " << total / students.size() << "\n";

    // Group by grade.
    std::map<int, std::vector<std::string>> by_grade;
    for (const auto& s : students) {
        by_grade[s.grade].push_back(s.name);
    }

    for (const auto& [grade, names] : by_grade) {
        std::cout << "grade " << grade << ": ";
        for (const auto& n : names) std::cout << n << " ";
        std::cout << "\n";
    }

    // Count above 90.
    auto above_90 = std::count_if(students.begin(), students.end(),
        [](const Student& s) { return s.score >= 90.0; });
    std::cout << "above 90: " << above_90 << "\n";

    return 0;
}
