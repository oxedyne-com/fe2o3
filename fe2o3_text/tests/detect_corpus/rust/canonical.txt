use std::collections::HashMap;
use std::io::{self, BufRead};

/// A simple key-value store backed by a hash map.
pub struct Store {
    data:     HashMap<String, String>,
    modified: bool,
}

impl Store {
    /// Create an empty store.
    pub fn new() -> Self {
        Self {
            data:     HashMap::new(),
            modified: false,
        }
    }

    /// Insert a key-value pair, returning the old value if present.
    pub fn insert(&mut self, key: String, value: String) -> Option<String> {
        self.modified = true;
        self.data.insert(key, value)
    }

    /// Retrieve a value by key.
    pub fn get(&self, key: &str) -> Option<&String> {
        self.data.get(key)
    }

    /// Return the number of entries.
    pub fn len(&self) -> usize {
        self.data.len()
    }

    /// Whether the store has been modified since creation.
    pub fn is_modified(&self) -> bool {
        self.modified
    }
}

fn main() {
    let mut store = Store::new();
    let stdin = io::stdin();

    for line in stdin.lock().lines() {
        let line = line.expect("IO error");
        let parts: Vec<&str> = line.splitn(2, '=').collect();
        if parts.len() == 2 {
            store.insert(parts[0].to_string(), parts[1].to_string());
        }
    }

    println!("Stored {} entries, modified: {}", store.len(), store.is_modified());
}
